class PointSet:

    points: List[Tuple[float, float]]
    
    def __init__(self, points: List[Tuple[float, float]]):
        self.points = points


    @staticmethod
    def parse_pointset(donnees_binaires: bytes) -> List[Tuple[float, float]]:
        if len(donnees_binaires) < 4:
            raise ValueError("données insuffisantes")

        (nombre_points,) = struct.unpack_from("<L", donnees_binaires, 0)
        taille_attendue = 4 + (8 * nombre_points)
        if len(donnees_binaires) != taille_attendue:
            raise ValueError("taille de données invalide")

        liste_points: List[Tuple[float, float]] = []
        decalage = 4
        for _ in range(nombre_points):
            x, y = struct.unpack_from("<ff", donnees_binaires, decalage)
            liste_points.append((x, y))
            decalage += 8

        return liste_points

    @staticmethod
    def serialize_pointset(points: Union['PointSet', List[Tuple[Any, Any]]]) -> bytes:
        if isinstance(points, PointSet):
            points_a_serialiser = points.points
        else:
            points_a_serialiser = points

        donnees = struct.pack("<L", len(points_a_serialiser))
        for point in points_a_serialiser:
            if not isinstance(point, (tuple, list)) or len(point) != 2:
                raise ValueError("format de point invalide")
            x, y = point
            if not isinstance(x, (int, float)) or not isinstance(y, (int, float)):
                raise TypeError("coordonnées invalides")
            donnees += struct.pack("<ff", float(x), float(y))

        return donnees
